7//! Round-Robin scheduler implementation for COCONUT-SVSM
8//!
9//! This module implements a round-robin scheduler for cooperative multi-tasking.
10//! It works by assigning a single owner for each struct [`Task`]. The owner
11//! depends on the state of the task:
12//!
13//! * [`PENDING`] A task has been created but has not yet been scheduled for
14//!   the first time, nor has it been placed into any wait queue or any other
15//!   type of queue.
16//! * [`RUNNING`] A task in running state is owned by the [`RunQueue`] and either
17//!   stored in the `run_list` (when the task is not actively running) or in
18//!   `current_task` when it is scheduled on the CPU.
19//! * [`BLOCKED`] A task in this state is waiting for an event to become runnable
20//!   again. It is owned by a wait object when in this state.
21//! * [`TERMINATED`] The task is about to be destroyed and owned by the [`RunQueue`].
22//!
23//! The scheduler is cooperative. A task runs until it voluntarily calls the
24//! [`schedule()`] function.
25//!
26//! Only when a task is in [`RUNNING`] or [`TERMINATED`] state it is assigned to a
27//! specific CPU. Tasks in the [`BLOCKED`] state have no CPU assigned and will run
28//! on the CPU where their event is triggered that makes them [`RUNNING`] again.
29//!
30//! [`PENDING`]: super::tasks::TaskState::PENDING
31//! [`RUNNING`]: super::tasks::TaskState::RUNNING
32//! [`BLOCKED`]: super::tasks::TaskState::BLOCKED
33//! [`TERMINATED`]: super::tasks::TaskState::TERMINATED

71/// A RunQueue implementation that uses an RBTree to efficiently sort the priority
72/// of tasks within the queue.
73#[derive(Debug, Default)]
74pub struct RunQueue {
75    /// Linked list with runable tasks
76    run_list: LinkedList<TaskRunListAdapter>,
77
78    /// Pointer to currently running task
79    current_task: Option<TaskPointer>,
80
81    /// Idle task - runs when there is no other runnable task
82    idle_task: Option<TaskPointer>,
83
84    /// Pointer to a task that should be woken when returning from idle
85    wake_from_idle: Option<TaskPointer>,
86}
87
88impl RunQueue {
89    /// Create a new runqueue for an id. The id would normally be set
90    /// to the APIC ID of the CPU that owns the runqueue and is used to
91    /// determine the affinity of tasks.
92    pub fn new() -> Self {
93        Self {
94            run_list: LinkedList::new(TaskRunListAdapter::new()),
95            current_task: None,
96            idle_task: None,
97            wake_from_idle: None,
98        }
99    }
100
101    /// Find the next task to run, which is either the task at the front of the
102    /// run_list or the idle task, if the run_list is empty.
103    ///
104    /// # Returns
105    ///
106    /// Pointer to next task to run
107    ///
108    /// # Panics
109    ///
110    /// Panics if there are no tasks to run and no idle task has been
111    /// allocated via [`set_idle_task()`](Self::set_idle_task).
112    fn get_next_task(&mut self) -> TaskPointer {
113        self.run_list
114            .pop_front()
115            .unwrap_or_else(|| self.idle_task.clone().unwrap())
116    }
117
118    /// Place a task back onto the run queue so it can be scheduled again.
119    fn enqueue_task(&mut self, task: TaskPointer) {
120        // Callers are expected to place the task into the RUNNING state
121        // before a task is queued.
122        debug_assert!(task.is_running());
123        if !task.is_idle_task() {
124            self.run_list.push_back(task);
125        }
126    }
127
128    /// Prepare to run a task by marking it runnable and placing it into the
129    /// run queue.
130    fn prepare_run_task(&mut self, task: TaskPointer) {
131        task.set_task_running();
132        self.enqueue_task(task);
133    }
134
135    /// Initializes the scheduler for this (RunQueue)[RunQueue]. This method is
136    /// called on the very first scheduling event when there is no current task
137    /// yet.  For consistency, it will always invoke the idle task using an
138    /// abbreviated task switch flow, and if there is another task that can
139    /// run, a full task switch will then occur.
140    ///
141    /// # Returns
142    ///
143    /// [TaskPointer] to the first task to run
144    pub fn schedule_init(&mut self) -> TaskPointer {
145        let task = self.idle_task.as_ref().unwrap().clone();
146        self.current_task = Some(task.clone());
147        task
148    }
149
150    /// Prepares a task switch. The function checks if a task switch needs to
151    /// be done and return pointers to the current and next task. It will
152    /// also call `enqueue_task()` on the current task in case a task-switch
153    /// is requested.
154    ///
155    /// # Parameters
156    ///
157    /// - `reschedule`: Indicates whether the current task being rescheduled.
158    ///   If so, it will be reinserted on the run list of the current
159    ///   processor.
160    ///
161    /// # Returns
162    ///
163    /// `None` when no task-switch is needed.
164    /// `Some` with current and next task in case a task-switch is required.
165    ///
166    /// # Panics
167    ///
168    /// Panics if there is no current task.
169    pub fn schedule_prepare(&mut self, reschedule: bool) -> Option<(TaskPointer, TaskPointer)> {
170        let current = if reschedule {
171            // Remove current and put it back into the RunQueue.  This is
172            // important to make sure the last runnable task keeps running,
173            // even if it calls schedule()
174            let current = self.current_task.take().unwrap();
175            self.enqueue_task(current.clone());
176            current
177        } else {
178            self.current_task.as_ref().unwrap().clone()
179        };
180
181        // Get next task and update current_task state
182        let next = self.get_next_task();
183        self.current_task = Some(next.clone());
184
185        // Check if task switch is needed
186        if current != next {
187            Some((current, next))
188        } else {
189            // A task switch is expected unless the current task is being
190            // rescheduled.
191            debug_assert!(reschedule);
192            None
193        }
194    }
195
196    pub fn current_task_id(&self) -> u32 {
197        self.current_task
198            .as_ref()
199            .map_or(INITIAL_TASK_ID, |t| t.get_task_id())
200    }
201
202    /// Sets the idle task for this RunQueue. This function sets a
203    /// OnceCell at the end and can thus be only called once.
204    ///
205    /// # Returns
206    ///
207    /// Ok(()) on success, SvsmError on failure
208    ///
209    /// # Panics
210    ///
211    /// Panics if the idle task was already set.
212    pub fn set_idle_task(&mut self, task: TaskPointer) {
213        task.set_idle_task();
214        task.set_task_running();
215
216        // Add idle task to global task list
217        TASKLIST.lock().list().push_front(task.clone());
218
219        self.idle_task.replace(task);
220    }
221
222    /// Gets a pointer to the idle task
223    ///
224    /// # Panics
225    ///
226    /// Panics if the idle task has not yet been set.
227    pub fn get_idle_task(&self) -> TaskPointer {
228        self.idle_task.as_ref().unwrap().clone()
229    }
230
231    /// Gets a pointer to the current task
232    ///
233    /// # Panics
234    ///
235    /// Panics if there is no current task.
236    pub fn current_task(&self) -> TaskPointer {
237        self.current_task.as_ref().unwrap().clone()
238    }
239
240    /// Wakes a task from idle if required.
241    ///
242    /// # Returns
243    ///
244    /// An `Option<TaskPointer>` indicating which task should be woken, if
245    /// any.
246    pub fn wake_from_idle(&mut self) -> Option<TaskPointer> {
247        self.wake_from_idle.take()
248    }
249}

251/// Global task list
252/// This contains every task regardless of affinity or run state.
253#[derive(Debug, Default)]
254pub struct TaskList {
255    list: Option<LinkedList<TaskListAdapter>>,
256}
257
258impl TaskList {
259    pub const fn new() -> Self {
260        Self { list: None }
261    }
262
263    pub fn list(&mut self) -> &mut LinkedList<TaskListAdapter> {
264        self.list
265            .get_or_insert_with(|| LinkedList::new(TaskListAdapter::new()))
266    }
267
268    pub fn get_task(&self, id: u32) -> Option<TaskPointer> {
269        let task_list = &self.list.as_ref()?;
270        let mut cursor = task_list.front();
271        while let Some(task) = cursor.get() {
272            if task.get_task_id() == id {
273                return cursor.clone_pointer();
274            }
275            cursor.move_next();
276        }
277        None
278    }
279
280    /// # Safety
281    /// The caller must ensure that `task` is already a member of this task
282    /// list.
283    unsafe fn terminate(&mut self, task: TaskPointer) -> Option<TaskPointer> {
284        // Set the task state as terminated. If the task being terminated is the
285        // current task then the task context will still need to be in scope until
286        // the next schedule() has completed. Schedule will keep a reference to this
287        // task until some time after the context switch.
288        let wakeup = task.set_task_terminated();
289        // SAFETY: `task` must be a task pointer that is part of this list.
290        let mut cursor = unsafe { self.list().cursor_mut_from_ptr(task.as_ref()) };
291        cursor.remove();
292
293        // Inform the caller about any task that may need to be woken.
294        wakeup
295    }
296}

300/// Creates, initializes and starts a new kernel task. Note that the task has
301/// already started to run before this function returns.
302///
303/// # Arguments
304///
305/// * entry: The function to run as the new tasks main function
306///
307/// # Returns
308///
309/// A new instance of [`TaskPointer`] on success, [`SvsmError`] on failure.
310pub fn start_kernel_task(
311    start_info: KernelThreadStartInfo,
312    name: String,
313) -> Result<TaskPointer, SvsmError> {
314    let cpu = this_cpu();
315    let task = Task::create(cpu, start_info, name)?;
316    TASKLIST.lock().list().push_back(task.clone());
317
318    // Put task on the runqueue of this CPU
319    cpu.runqueue_mut().prepare_run_task(task.clone());
320
321    schedule();
322
323    Ok(task)
324}

326/// Creates, initializes and starts a new kernel thread of the currently
327/// running kernel task. Note that the thread has already started to run before
328/// this function returns.
329///
330/// # Arguments
331///
332/// * `entry` -  The function to run as the new task's main function
333/// * `start_parameter` - Parameter of type `usize` to pass to the new threads main function.
334///
335/// # Returns
336///
337/// A new instance of [`TaskPointer`] on success, [`SvsmError`] on failure.
338pub fn start_kernel_thread(start_info: KernelThreadStartInfo) -> Result<TaskPointer, SvsmError> {
339    let current_task = current_task();
340    let cpu = this_cpu();
341    let task = Task::create_thread(
342        cpu,
343        start_info,
344        current_task.get_task_name().clone(),
345        current_task,
346    )?;
347    TASKLIST.lock().list().push_back(task.clone());
348
349    // Put task on the runqueue of this CPU
350    cpu.runqueue_mut().prepare_run_task(task.clone());
351
352    schedule();
353
354    Ok(task)
355}

402/// Waits for a task to terminate
403pub fn wait_for_termination(task: TaskPointer) {
404    // Waiting may require a task switch, so ensure that this is safe.
405    preemption_checks();
406
407    // Prepare a wait state based on the current execution state of the target
408    // task.
409    let wait_result = task.wait_for_exit();
410
411    // If a wait is required, then switch to a different task until the wait
412    // can be satisfied.
413    if let Some(guard) = wait_result {
414        select_new_task(false, Some(guard));
415    }
416}

418/// Terminates the current task.
419///
420/// # Panic
421///
422/// This function must only be called after scheduling is initialized, otherwise it will panic.
423fn current_task_terminated() {
424    let cpu = this_cpu();
425    let mut rq = cpu.runqueue_mut();
426
427    let task_node = rq
428        .current_task
429        .as_mut()
430        .expect("Task termination handler called when there is no current task");
431    // SAFETY: the scheduler guarantees that `current_task` always points to a
432    // valid task, and every task has its pointer pushed into the global task
433    // list during its creation.
434    let wakeup = unsafe { TASKLIST.lock().terminate(task_node.clone()) };
435
436    // If another thread must be woken as a result of the termination, then
437    // schedule it now.
438    if let Some(wake_task) = wakeup {
439        rq.prepare_run_task(wake_task);
440    }
441}

443pub fn terminate() -> ! {
444    // Terminating a task will result in a task change, so preemption must
445    // be allowable.
446    preemption_checks();
447    current_task_terminated();
448
449    // The current task will not run again, so switch to a different task.
450    select_new_task(false, None);
451    unreachable!("terminated task rescheduled");
452}

454pub fn go_idle() {
455    // Entering an idle state will result in a task change, so preemption must
456    // be allowable.
457    preemption_checks();
458
459    // Mark this task as blocked and indicate that it is waiting for wake after
460    // idle.  Only one task on each CPU can be in the wake-from-idle state at
461    // one time.
462    let task = this_cpu().current_task();
463    task.set_task_blocked();
464    let mut runqueue = this_cpu().runqueue_mut();
465    assert!(runqueue.wake_from_idle.is_none());
466    runqueue.wake_from_idle = Some(task);
467    drop(runqueue);
468
469    // Find another task to run.  If no other task is runnable, then the idle
470    // thread will execute.
471    select_new_task(false, None);
472}

474pub fn set_affinity(cpu_index: usize) {
475    // Changes to affinity mak cause a scheduling change, so verify that
476    // scheduling operations are safe.
477    preemption_checks();
478
479    // Affinity signaling is only required if the target CPU is not the current
480    // CPU.
481    if cpu_index != this_cpu().get_cpu_index() {
482        let task = this_cpu().current_task();
483        let target_cpu = PERCPU_AREAS.get_by_cpu_index(cpu_index);
484
485        // Disable interrupts to prevent delays in scheduling once the task
486        // has been queued on the target processor.
487        let guard = IrqGuard::new();
488
489        // Join this task to the run queue of the target CPU.
490        target_cpu.runqueue_mut().enqueue_task(task);
491
492        // Send a scheduler interrupt to the target CPU so that if it is idle,
493        // it wakes and runs this task.
494        let icr = ApicIcr::new()
495            .with_vector(SCHEDULE_VECTOR as u8)
496            .with_destination(target_cpu.apic_id());
497        apic_post_irq(icr.into());
498
499        // Find another task to run.  The scheduler will complete the affinity
500        // change once a new task has been selected on this processor.
501        select_new_task(false, Some(guard));
502    }
503}

505// SAFETY: the caller is required to guarantee that the incoming task is
506// referenced somewhere (at least the current CPU's run queue current task
507// pointer) or else it will be destroyed before it is entered.
508#[inline(always)]
509unsafe fn switch_to(prev_task: Option<TaskPointer>, next_task: TaskPointer) -> Option<TaskPointer> {
510    // Capture a pointer to the new task and consume its `Arc`.  The caller
511    // guarantees that the task will still exist.
512    let next = Arc::as_ptr(&next_task);
513    drop(next_task);
514
515    // Consume the `Arc` describing the currently executing task without
516    // adjusting the reference count.  This will ensure that the current task
517    // is not fully dereferenced while it is still executing.  The reference
518    // will be rebalanced after the task switch.
519    let prev = if let Some(task) = prev_task {
520        Arc::into_raw(task)
521    } else {
522        null_mut()
523    };
524
525    // SAFETY: Assuming the caller has provided the correct task pointers,
526    // the page table and stack information in those tasks are correct and
527    // can be used to switch to the correct page table and execution stack.
528    unsafe {
529        let cr3 = (*next).page_table.lock().cr3_value().bits();
530
531        // Switch to new task
532        let new_prev = switch_context(
533            prev as usize,
534            next as usize,
535            ptr::from_ref(this_cpu()) as usize,
536            cr3,
537        );
538        complete_task_switch(new_prev)
539    }
540}

542/// # Safety
543/// The caller must guarantee that the task pointer argument is the one that
544/// was returned from the context switcher.
545pub unsafe fn complete_task_switch(prev: usize) -> Option<TaskPointer> {
546    // If there was a previous task, then create a new `Arc` that describes
547    // the previous task.  This will balance the reference count that was
548    // not decremented when the `Arc` was consumed prior to the task
549    // switch.
550    if prev != 0 {
551        // SAFETY: the caller guarantees that the task pointer is valid.
552        unsafe { Some(Arc::from_raw(prev as *const Task)) }
553    } else {
554        None
555    }
556}

558/// Initializes the [RunQueue] on the current CPU. It will switch to the idle
559/// task and initialize the current_task field of the RunQueue. After this
560/// function has ran it is safe to call [`schedule()`] on the current CPU.
561///
562/// # Safety
563///
564/// This function can only be called when it is known that there is no current
565/// task.  Otherwise, the run state can become corrupted, and thus future
566/// calculation of task pointers can be incorrect.
567pub unsafe fn schedule_init() {
568    let guard = IrqGuard::new();
569    // SAFETY: The caller guarantees that there is no current task, and the
570    // pointer obtained for the next task will always be correct, thus
571    // providing a guarantee that the task switch will be safe.
572    let prev_task = unsafe { switch_to(None, this_cpu().schedule_init()) };
573
574    // Drop the interrupt guard before allowing the previous task reference to
575    // go out of scope.  This ensures that the task destructor will run with
576    // interrupts enabled.
577    drop(guard);
578    drop(prev_task);
579}

581/// Enters an idle state if there is no task that can run.
582pub fn scheduler_idle() {
583    // All decisions must be made with interrupts disabled to ensure that the
584    // scheduler state does not change before committing to go idle.
585    let guard = IrqGuard::new();
586
587    let queue_empty = this_cpu().runqueue_mut().run_list.front().is_null();
588
589    if queue_empty {
590        SVSM_PLATFORM.idle_halt(&guard);
591    }
592}

594fn preemption_checks() {
595    assert!(irq_nesting_count() == 0);
596    assert!(raw_get_tpr() == 0 || !SVSM_PLATFORM.use_interrupts());
597}

599/// # Safety
600/// The caller must guarantee that this is only called on a valid task pointer
601/// that is actively scheduled on the current CPU.  Called only from the
602/// task switch code.
603#[unsafe(no_mangle)]
604pub unsafe fn update_task_percpu_page_tables(t: *const Task) {
605    // SAFETY: the caller guarantees the correctness of the task pointer.
606    let task = unsafe { &*t };
607    let mut pt = task.page_table.lock();
608    this_cpu().populate_page_table(&mut pt);
609}

611/// Perform a task switch and hand the CPU over to the next task on the
612/// run-list. In case the current task is terminated, it will be destroyed after
613/// the switch to the next task.
614pub fn schedule() {
615    // check if preemption is safe
616    preemption_checks();
617
618    select_new_task(true, None);
619}

621/// Select another task to run.  If rescheduling is requested, the current
622/// task will be placed back on the current processor's run queue so it can
623/// be eligible to run again.
624fn select_new_task(reschedule: bool, irq_guard: Option<IrqGuard>) {
625    // If the caller has not already disabled interrupts, then disable them
626    // now.
627    let guard = irq_guard.unwrap_or_default();
628
629    let work = this_cpu().schedule_prepare(reschedule);
630
631    // !!! Runqueue lock must be released here !!!
632    let prev_task = if let Some((current, next)) = work {
633        // Ensure that the current stack bounds of the current CPU are adjusted
634        // to reflect the task being scheduled.
635        this_cpu().set_current_stack(next.stack_bounds());
636
637        // SAFETY: ths stack pointer is known to be correct.
638        unsafe {
639            this_cpu().set_tss_rsp0(next.stack_bounds.end());
640        }
641        if is_cet_ss_enabled() {
642            // SAFETY: Task::exception_shadow_stack is always initialized when
643            // creating a new Task.
644            unsafe {
645                write_msr(PL0_SSP, next.shadow_stack_base.bits() as u64);
646            }
647        }
648
649        // Get task-pointers.
650        //
651        // SAFETY: the scheduler guarantees that both `current` and `next`
652        // always point to valid tasks. The XSAVE area in each task must be
653        // valid and not aliased.
654        unsafe {
655            // Capture a pointer to the current task.  This pointer can be
656            // decoupled from the lifetime of the `Arc` because it is valid
657            // as long as the task continues to execute.
658            let current_ptr = Arc::as_ptr(&current);
659            sse_save_context(u64::from((*current_ptr).xsa.vaddr()));
660
661            // Switch tasks.  This call must consume the `Arc` references to
662            // both tasks because this call might never return (if the current
663            // task is being terminated) and therefore no references can live
664            // beyond this call.  This call will return an `Arc` reference to
665            // the task that ran most recently, which may be the final
666            // reference to the task if the task was terminated.
667            let prev_task = switch_to(Some(current), next);
668
669            // The previously captured task pointer is known to be valid
670            // because the task is still executing.
671            sse_restore_context(u64::from((*current_ptr).xsa.vaddr()));
672
673            prev_task
674        }
675    } else {
676        None
677    };
678
679    // Drop the interrupt guard before allowing the previous task reference to
680    // go out of scope.  This ensures that the task destructor will run with
681    // interrupts enabled.
682    drop(guard);
683    drop(prev_task);
684}

686pub fn wake_and_schedule_task(task: TaskPointer) {
687    debug_assert!(!task.is_running());
688    this_cpu().runqueue_mut().prepare_run_task(task);
689    schedule();
690}

692unsafe extern "C" {
693    fn switch_context(prev: usize, next: usize, this_cpu: usize, cr3: usize) -> usize;
694}
695
696global_asm!(
697    r#"
698        .section .text
699
700    switch_context:
701        // Arguments:
702        // rdi: previous task pointer
703        // rsi: new task pointer
704        // rdx: current per-CPU pointer
705        // rcx: paging root of the new task
706        //
707        // Save the current context. The layout must match the TaskContext
708        // structure.  Only callee-save registers need to be pushed here; the
709        // remainder of the TaskContext frame can simply be allocated on the
710        // stack.
711        pushq   %rbp
712        pushq   %rbx
713        pushq   %r15
714        pushq   %r14
715        pushq   %r13
716        pushq   %r12
717        subq    $24, %rsp
718
719        // If `prev` is not null...
720        testq   %rdi, %rdi
721        // The initial stack is always mapped in the new page table.
722        jz      1f
723
724        // Save the current stack pointer
725        movq    %rsp, {TASK_RSP_OFFSET}(%rdi)
726
727        // Switch to a stack pointer that's valid in both the old and new page
728        // tables.
729        mov     {CONTEXT_SWITCH_RSP_OFFSET}(%rdx), %rsp
730
731        // Clear the frame pointer since it is no longer meaningful.
732        xorl    %ebp, %ebp
733
734        // Switch shadow stacks if required.
735        cmpb    $0, {IS_CET_ENABLED}(%rip)
736        je      4f
737        // Save the current shadow stack pointer
738        rdssp   %rax
739        sub     $8, %rax
740        movq    %rax, {TASK_SSP_OFFSET}(%rdi)
741        // Switch to a shadow stack that's valid in both page tables and move
742        // the "shadow stack restore token" to the old shadow stack.
743        mov     ${CONTEXT_SWITCH_RESTORE_TOKEN}, %rax
744        rstorssp (%rax)
745        saveprevssp
746
747    4:
748        // Switch to the current CPU's page table to ensure that the page
749        // table remains correct for the current CPU even if the previous task
750        // is scheduled onto another CPU and has its per-CPU address space
751        // updated.
752        movq    {PERCPU_PGTBL_OFFSET}(%rdx), %rax
753        movq    %rax, %cr3
754
755        // Mark the previous task as inactive.  This must be done after
756        // switching off of its stack because as soon as it is marked as
757        // inactive, another processor is free to immediately switch to that
758        // thread's stack.
759        andb    $0, {TASK_STATE_ACTIVE}(%rdi)
760
761    1:
762        // Switch to the new task state.
763
764        // Wait until the new task is inactive.  It may still be running
765        // on another processor so its stack cannot be consumed until its
766        // stack is no longer active on any processor.
767    3:
768        pause
769        movb    $1, %al
770        lock xchgb {TASK_STATE_ACTIVE}(%rsi), %al
771        testb   %al, %al
772        jnz     3b
773
774        // Check to see whether the task is moving across CPUs.  If so, its
775        // per-CPU page table state must be updated.
776        movq    {PERCPU_SHARED_OFFSET}(%rdx), %r8
777        movq    {PERCPU_SHARED_INDEX_OFFSET}(%r8), %rax
778        cmpq    {TASK_CPU_OFFSET}(%rsi), %rax
779        jz      5f
780        movq    %rax, {TASK_CPU_OFFSET}(%rsi)
781
782        // Save local registers before calling out to do the page table update.
783        // Save only the registers that will be needed following the update,
784        // and ensure that a multiple of 16 bytes is pushed to maintain
785        // compliance with the stack ABI requirement.
786        pushq   %rsi
787        pushq   %rdi
788        pushq   %rcx
789        subq    $8, %rsp
790
791        movq    %rsi, %rdi
792        call    update_task_percpu_page_tables
793
794        addq    $8, %rsp
795        popq    %rcx
796        popq    %rdi
797        popq    %rsi
798
799    5:
800        // Switch to the new task page tables
801        movq    %rcx, %cr3
802
803        cmpb    $0, {IS_CET_ENABLED}(%rip)
804        je      2f
805        // Switch to the new task shadow stack and move the "shadow stack
806        // restore token" back.
807        mov     {TASK_SSP_OFFSET}(%rsi), %rax
808        rstorssp (%rax)
809        saveprevssp
810
811    2:
812        // Switch to the new task stack
813        movq    {TASK_RSP_OFFSET}(%rsi), %rsp
814
815        // Pass the previous task pointer (if any) back to the caller.  This
816        // is done both as a return value (if this function was called from
817        // the task switcher) and as the first parameter (if this routine
818        // will return to the task entry point.
819        movq    %rdi, %rax
820
821        // Restore the task state, following the layout of TaskContext.
822        popq    %rsi
823        popq    %rdx
824        popq    %rcx
825        popq    %r12
826        popq    %r13
827        popq    %r14
828        popq    %r15
829        popq    %rbx
830        popq    %rbp
831
832        ret
833    "#,
834    TASK_RSP_OFFSET = const offset_of!(Task, rsp),
835    TASK_SSP_OFFSET = const offset_of!(Task, ssp),
836    TASK_STATE_ACTIVE = const TASK_ACTIVE_OFFSET,
837    TASK_CPU_OFFSET = const TASK_CUR_CPU_OFFSET,
838    IS_CET_ENABLED = sym IS_CET_ENABLED,
839    CONTEXT_SWITCH_RSP_OFFSET = const PERCPU_CTXT_SWITCH_STACK_OFFSET,
840    PERCPU_SHARED_OFFSET = const PERCPU_SHARED_OFFSET,
841    PERCPU_SHARED_INDEX_OFFSET = const PERCPU_SHARED_INDEX_OFFSET,
842    PERCPU_PGTBL_OFFSET = const PERCPU_PAGING_ROOT_OFFSET,
843    CONTEXT_SWITCH_RESTORE_TOKEN = const CONTEXT_SWITCH_RESTORE_TOKEN.as_usize(),
844    options(att_syntax)
845);

847/// The location of a cpu-local shadow stack restore token that's mapped into
848/// every set of page tables for use during context switches.
849///
850/// One interesting difference between the normal stack pointer and the shadow
851/// stack pointer is how they can be switched: For the normal stack pointer we
852/// can just move a new value into the RSP register. This doesn't work for the
853/// SSP register (the shadow stack pointer) because there's no way to directly
854/// move a value into it. Instead we have to use the `rstorssp` instruction.
855/// The key difference between this instruction and a regular `mov` is that
856/// `rstorssp` expects a "shadow stack restore token" to be at the top of the
857/// new shadow stack (this is just a special value that marks the top of a
858/// inactive shadow stack). After switching to a new shadow stack, the previous
859/// shadow stack is now inactive, and so the `saveprevssp` instruction can be
860/// used to transfer the shadow stack restore token from the new shadow stack
861/// to the previous one: `saveprevssp` atomically pops the stack token of the
862/// new shadow stack and pushes it on the previous shadow stack. This means
863/// that we have to execute both `rstorssp` and `saveprevssp` every time we
864/// want to switch the shadow stacks.
865///
866/// There's one major problem though: `saveprevssp` needs to access both the
867/// previous and the new shadow stack, but we only map each shadow stack into a
868/// single task's page tables. If each set of page tables only has access to
869/// either the previous or the new shadow stack, but not both, we can't execute
870/// `saveprevssp` and so we we can't move the shadow stack restore token to the
871/// previous shadow stack. If there's no shadow stack restore token on the
872/// previous shadow stack that means we can't restore this shadow stack at a
873/// later point. To work around this, we map another shadow stack into each
874/// CPU's set of pagetables. This allows us to do the following:
875///
876/// 1. Switch to the context-switch shadow stack using `rstorssp`.
877/// 2. Transfer the shadow stack restore token from the context switch shadow
878///    stack to the previous shadow stack by executing `saveprevssp`.
879/// 3. Switch the page tables. This doesn't lead to problems with the context
880///    switch shadow stack because it's mapped into both page tables.
881/// 4. Switch to the new shadow stack using `rstorssp`.
882/// 5. Transfer the shadow stack restore token from the new shadow stack back
883///    to the context switch shadow stacks by executing `saveprevssp`.
884///
885/// We just switched between two shadow stack tables in different page tables :)
886///
887/// Stack offset calculation:
888///
889/// 0x1ff8 = Size(GuardPage) + Size(ShadowStack) - 8; where Size(GuardPage) == Size(ShadowStack) == PAGE_SIZE.
890const CONTEXT_SWITCH_RESTORE_TOKEN: VirtAddr = SVSM_CONTEXT_SWITCH_SHADOW_STACK.const_add(0x1ff8);
