14static CPUID_PAGE: ImmutAfterInitCell<&SnpCpuidTable> = ImmutAfterInitCell::uninit();
15
16const _: () = assert!(size_of::<SnpCpuidTable>() <= PAGE_SIZE);
17
18pub fn register_cpuid_table(table: &'static SnpCpuidTable) {
19    CPUID_PAGE
20        .init(table)
21        .expect("Could not initialize CPUID page");
22}

24/// # Safety
25/// The caller must specify a valid virtual address to use for CPUID table
26/// initialization.
27///
28/// # Panics
29///
30/// Panics if the provided address is not aligned to a [`SnpCpuidTable`].
31pub unsafe fn init_cpuid_table(addr: VirtAddr) {
32    // SAFETY: the caller takes responsibility for the correctness of the
33    // virtual address.
34    let table = unsafe {
35        addr.aligned_mut::<SnpCpuidTable>()
36            .expect("Misaligned SNP CPUID table address")
37    };
38
39    for func in table.entries_mut() {
40        if func.eax_in == 0x8000001f {
41            func.eax_out |= 1 << 28;
42        }
43    }
44
45    register_cpuid_table(table);
46}
47
48/// Copy a CPUID page's content to memory pointed to by a [`VirtAddr`]
49///
50/// # Safety
51///
52/// The caller should verify that `dst` points to mapped memory whose size is
53/// at least 4K. We assert above at compile time that SnpCpuidTable fits within
54/// a page, so the write is safe.
55///
56/// The caller should verify not to corrupt arbitrary memory, as this function
57/// doesn't make any checks in that regard.
58pub unsafe fn copy_cpuid_table_to(dst: VirtAddr) {
59    let start = dst.as_mut_ptr::<u8>();
60    // SAFETY: caller must ensure the address is valid and not aliased.
61    unsafe {
62        // Zero target and copy data
63        start.write_bytes(0, PAGE_SIZE);
64        start
65            .cast::<SnpCpuidTable>()
66            .copy_from_nonoverlapping(*CPUID_PAGE, 1);
67    }
68}

70#[derive(Clone, Copy, Debug)]
71#[repr(C, packed)]
72pub struct CpuidLeaf {
73    pub cpuid_fn: u32,
74    pub cpuid_subfn: u32,
75    pub eax: u32,
76    pub ebx: u32,
77    pub ecx: u32,
78    pub edx: u32,
79}
80
81impl CpuidLeaf {
82    pub fn new(cpuid_fn: u32, cpuid_subfn: u32) -> Self {
83        CpuidLeaf {
84            cpuid_fn,
85            cpuid_subfn,
86            eax: 0,
87            ebx: 0,
88            ecx: 0,
89            edx: 0,
90        }
91    }
92}
93
94pub fn cpuid_table_raw(eax: u32, ecx: u32, xcr0: u64, xss: u64) -> Option<CpuidResult> {
95    CPUID_PAGE
96        .entries()
97        .iter()
98        .find(|f| eax == f.eax_in && ecx == f.ecx_in && xcr0 == f.xcr0_in && xss == f.xss_in)
99        .map(|f| CpuidResult {
100            eax: f.eax_out,
101            ebx: f.ebx_out,
102            ecx: f.ecx_out,
103            edx: f.edx_out,
104        })
105}
106
107pub fn cpuid_table(eax: u32, ecx: u32) -> Option<CpuidResult> {
108    cpuid_table_raw(eax, ecx, 0, 0)
109}

111pub fn dump_cpuid_table() {
112    if let Ok(table) = CPUID_PAGE.try_get_inner() {
113        let entries = table.entries();
114        log::trace!("CPUID Table entry count: {}", entries.len());
115
116        for func in entries {
117            let eax_in = func.eax_in;
118            let ecx_in = func.ecx_in;
119            let xcr0_in = func.xcr0_in;
120            let xss_in = func.xss_in;
121            let eax_out = func.eax_out;
122            let ebx_out = func.ebx_out;
123            let ecx_out = func.ecx_out;
124            let edx_out = func.edx_out;
125            log::trace!(
126                "EAX_IN: {eax_in:#010x} ECX_IN: {ecx_in:#010x} XCR0_IN: {xcr0_in:#010x} XSS_IN: {xss_in:#010x} EAX_OUT: {eax_out:#010x} EBX_OUT: {ebx_out:#010x} ECX_OUT: {ecx_out:#010x} EDX_OUT: {edx_out:#010x}"
127            );
128        }
129    }
130}
